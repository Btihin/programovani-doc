```tsx showLineNumbers {10-12} copy filename="Citac.tsx"
function MojeTlacitko() {
  const [pocet, setPocet] = useState(0);

  function handleClick() {
    setPocet(pocet + 1);
  }

  return (
    <div>
      <button onClick={handleClick} className={styles.counter}>
        Kliknuto {pocet} krát.
      </button>
    </div>
  );
}
```

}
